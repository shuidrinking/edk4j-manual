5.5.2、异常常量类ExceptionConstant
异常常量类：com.edk4j.core.constant.ExceptionConstant
抛出异常的时候，不要每次都new一个，你可以在你的action里提前new一个异常对象作为类变量用来抛出。
也可以从异常常量类里酌情选择合适的异常常量抛出去。
ExceptionConstant类中的异常常量清单： 
	TranFailException TRAN_FAIL_EXCEPTION=new TranFailException("交易失败");
	TranFailException VERIFYCODE_EXCEPTION = new TranFailException(BaseErrorCode.VERIFYCODE_ERROR,"验证码错误");
	DuplicateSubmitException DUMPLICATE_SUBMIT_EXCEPTION = new DuplicateSubmitException("不能重复提交");
	SessionTimeOutException SESSION_TIMEOUT_EXCEPTION = new SessionTimeOutException("登录已超时");
	SystemComponentException SYSTEM_COMPONENT_EXCEPTION = new SystemComponentException("系统错误");
	TranLogException TRAN_LOG_EXCEPTION = new TranLogException("记录运行日志异常");
	PorpertiesCopyException PROPERTIES_COPY_EXCEPTION = new PorpertiesCopyException("属性值复制异常");
	CacheException CACHE_EXCEPTION = new CacheException("缓存异常");
	CacheException CACHE_SERVER_EXCEPTION = new CacheException(BaseErrorCode.USER_DEFINED_ERROR, new String[]{"本地缓存配置或缓存服务器不可用"});
	CacheException NO_SESSION_CACHED_EXCEPTION = new CacheException(BaseErrorCode.NO_SESSION_CACHED_ERROR, "无法获取会话数据");
	FakeDataException FAKE_DATA_EXCEPTION = new FakeDataException("伪造交易数据");
	IllegalVisitException ILLEGAL_VISIT_EXCEPTION = new IllegalVisitException("非法访问");
	RequestException INVALID_REQUEST_EXCEPTION = new RequestException("无效请求");
