/**
 * 检查非法值的正则表达式
 */
public static Pattern invalidValuePattern = Pattern.compile(BaseConstant.REQUEST_PARAMETER_SAFECHECK_PATTERN, Pattern.CASE_INSENSITIVE);
/**
 * 验证入参的对象中是否安全
 * 注意：不要用这个方法去检查文件或者流对象
 * @param paramObject 可以为form中的表单值，或者入参的VO，或者入参中的集合
 */
public abstract boolean isInputObjectSafe(Object paramObject) ;
/**
 * 检查request参数，在方法内自定义进行解析检查
 * @param request
 */
public abstract boolean isRequestSafe(HttpServletRequest request);
/**
 * 对单个字段值实施验证
 * @param valueString
 * @return
 */
public abstract boolean isStringValueSafe(String valueString);
